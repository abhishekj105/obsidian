import hashlib, json, os, csv
from datetime import datetime
from pathlib import Path

import requests
from bs4 import BeautifulSoup
import pdfplumber
import pandas as pd
from apscheduler.schedulers.blocking import BlockingScheduler

# ─── CONFIG ──────────────────────────────────────────────────
DATA_DIR = Path("data")
DATA_DIR.mkdir(exist_ok=True)

PRICE_CSV   = DATA_DIR / "polymer_prices.csv"
TRIGGER_FILE= DATA_DIR / "trigger.json"
HASH_FILE   = DATA_DIR / "last_hashes.json"

HEADERS = {"User-Agent": "Mozilla/5.0 (compatible; WiseIntel/1.0)"}

# ─── HASH HELPERS ────────────────────────────────────────────
def load_hashes():
    if HASH_FILE.exists():
        return json.loads(HASH_FILE.read_text())
    return {}

def save_hashes(h):
    HASH_FILE.write_text(json.dumps(h, indent=2))

def content_hash(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()[:16]

# ─── CSV WRITER ──────────────────────────────────────────────
def append_price(source, polymer, grade, price_inr, pct_change=None, score=1):
    row = {
        "timestamp":   datetime.now().isoformat(),
        "source":      source,
        "polymer":     polymer,
        "grade":       grade,
        "price_inr":   price_inr,
        "pct_change":  round(pct_change, 2) if pct_change else None,
        "signal_score":score,
    }
    write_header = not PRICE_CSV.exists()
    with open(PRICE_CSV, "a", newline="") as f:
        w = csv.DictWriter(f, fieldnames=row.keys())
        if write_header:
            w.writeheader()
        w.writerow(row)
    print(f"  [+] {source} | {polymer} {grade} | ₹{price_inr:,} | score={score}")
    return row

def compute_pct_change(source, polymer, grade, new_price):
    """Compare new_price against 7-day average from CSV."""
    if not PRICE_CSV.exists():
        return 0.0
    df = pd.read_csv(PRICE_CSV)
    df = df[(df.source == source) & (df.polymer == polymer) & (df.grade == grade)]
    if df.empty:
        return 0.0
    df["timestamp"] = pd.to_datetime(df["timestamp"])
    week_ago = pd.Timestamp.now() - pd.Timedelta(days=7)
    recent = df[df.timestamp >= week_ago]["price_inr"]
    if recent.empty:
        return 0.0
    avg = recent.mean()
    return ((new_price - avg) / avg) * 100

def score_signal(pct_change: float, is_ioc_revision=False, is_force_majeure=False) -> int:
    if is_force_majeure:
        return 5
    if is_ioc_revision and abs(pct_change) > 3:
        return 5
    if abs(pct_change) > 4:
        return 5
    if is_ioc_revision or abs(pct_change) > 2:
        return 4
    if abs(pct_change) > 1:
        return 3
    return 1

def maybe_trigger(score, signal_data):
    if score >= 3:
        payload = {**signal_data, "score": score, "triggered_at": datetime.now().isoformat()}
        TRIGGER_FILE.write_text(json.dumps(payload, indent=2))
        print(f"  [!] TRIGGER written — score {score}: {signal_data['polymer']} {signal_data.get('pct_change', '')}%")

# ─── FETCHER 1: PLASTEMART (aggregates RIL + IOC + HPCL) ─────
def fetch_plastemart():
    """
    Plastemart.com publishes a daily polymer price board — PP, PE, PVC
    from all major Indian producers. Updated ~10am and 3pm IST.
    """
    print("[Plastemart] fetching...")
    hashes = load_hashes()
    url = "https://www.plastemart.com/plastic-technical-article/plastics-raw-material-prices-in-india"

    try:
        r = requests.get(url, headers=HEADERS, timeout=15)
        r.raise_for_status()
    except Exception as e:
        print(f"  [Plastemart] ERROR: {e}")
        return

    h = content_hash(r.content)
    if hashes.get("plastemart") == h:
        print("  [Plastemart] no change")
        return
    hashes["plastemart"] = h
    save_hashes(hashes)

    soup = BeautifulSoup(r.text, "lxml")

    # Find the price tables — Plastemart uses <table> tags with polymer names
    # This selector targets their main price board table
    for table in soup.find_all("table"):
        rows = table.find_all("tr")
        for row in rows:
            cells = [td.get_text(strip=True) for td in row.find_all("td")]
            if len(cells) < 3:
                continue
            name = cells[0].upper()
            # Match known polymers
            polymer, grade = None, "general"
            if "PP" in name or "POLYPROPYLENE" in name:
                polymer = "PP"
            elif "HDPE" in name:
                polymer, grade = "PE", "HDPE"
            elif "LLDPE" in name:
                polymer, grade = "PE", "LLDPE"
            elif "LDPE" in name:
                polymer, grade = "PE", "LDPE"
            elif "PVC" in name:
                polymer = "PVC"
            if not polymer:
                continue

            # Price is usually in the 2nd or 3rd cell — try to parse ₹
            for cell in cells[1:]:
                clean = cell.replace(",", "").replace("₹", "").replace("Rs", "").strip()
                try:
                    price = float(clean)
                    if 50_000 < price < 300_000:  # sanity: ₹50k–₹3L/MT is valid range
                        pct = compute_pct_change("plastemart", polymer, grade, price)
                        sc  = score_signal(pct)
                        row_data = append_price("plastemart", polymer, grade, price, pct, sc)
                        maybe_trigger(sc, row_data)
                        break
                except ValueError:
                    continue

# ─── FETCHER 2: IOC POLYMER PRICE PDFs ───────────────────────
def fetch_ioc():
    """
    Indian Oil Corporation publishes price revision PDFs.
    We check their polymer page for new PDF links, download and parse.
    """
    print("[IOC] fetching price notifications...")
    hashes = load_hashes()
    url = "https://iocl.com/products/polymers"

    try:
        r = requests.get(url, headers=HEADERS, timeout=15)
        r.raise_for_status()
    except Exception as e:
        print(f"  [IOC] ERROR: {e}")
        return

    soup = BeautifulSoup(r.text, "lxml")
    # Find all PDF links on the page
    pdf_links = [
        a["href"] for a in soup.find_all("a", href=True)
        if a["href"].lower().endswith(".pdf")
        and any(kw in a.get_text().lower() + a["href"].lower()
                for kw in ["price", "polymer", "pp", "pe", "pvc"])
    ]

    for link in pdf_links[:5]:  # check last 5 PDF links only
        full_url = link if link.startswith("http") else f"https://iocl.com{link}"
        link_hash = content_hash(full_url.encode())
        if hashes.get(f"ioc_pdf_{link_hash}") == "seen":
            continue  # already processed this PDF

        try:
            pdf_r = requests.get(full_url, headers=HEADERS, timeout=20)
            pdf_r.raise_for_status()
        except Exception as e:
            print(f"  [IOC PDF] ERROR downloading {full_url}: {e}")
            continue

        # Save temporarily
        tmp_pdf = DATA_DIR / "ioc_latest.pdf"
        tmp_pdf.write_bytes(pdf_r.content)
        hashes[f"ioc_pdf_{link_hash}"] = "seen"
        save_hashes(hashes)

        # Parse with pdfplumber
        prices_found = parse_ioc_pdf(tmp_pdf)
        for p in prices_found:
            pct = compute_pct_change("ioc", p["polymer"], p["grade"], p["price"])
            sc  = score_signal(pct, is_ioc_revision=True)
            row_data = append_price("ioc", p["polymer"], p["grade"], p["price"], pct, sc)
            maybe_trigger(sc, row_data)

def parse_ioc_pdf(pdf_path: Path) -> list:
    """Extract polymer prices from IOC price notification PDF."""
    results = []
    try:
        with pdfplumber.open(pdf_path) as pdf:
            for page in pdf.pages:
                tables = page.extract_tables()
                for table in tables:
                    for row in table:
                        if not row:
                            continue
                        row_text = " ".join(str(c) for c in row if c).upper()
                        polymer, grade = None, "general"
                        if "POLYPROPYLENE" in row_text or " PP " in row_text:
                            polymer = "PP"
                            if "HOMO" in row_text:
                                grade = "homopolymer"
                            elif "COPOL" in row_text:
                                grade = "copolymer"
                        elif "HDPE" in row_text:
                            polymer, grade = "PE", "HDPE"
                        elif "LLDPE" in row_text:
                            polymer, grade = "PE", "LLDPE"
                        elif "PVC" in row_text:
                            polymer = "PVC"
                        if not polymer:
                            continue
                        # Look for price — typically a 5–6 digit number
                        for cell in row:
                            if not cell:
                                continue
                            clean = str(cell).replace(",", "").strip()
                            try:
                                price = float(clean)
                                if 50_000 < price < 300_000:
                                    results.append({"polymer": polymer, "grade": grade, "price": price})
                                    break
                            except ValueError:
                                continue
    except Exception as e:
        print(f"  [IOC PDF parse] ERROR: {e}")
    return results

# ─── FETCHER 3: Reuters RSS — geopolitical triggers ──────────
KEYWORDS = [
    "hormuz", "naphtha", "polymer", "polypropylene",
    "polyethylene", "pvc", "feedstock", "force majeure",
    "iran", "strait", "crude oil"
]

def fetch_reuters_rss():
    """
    Reuters commodity RSS — parse titles for keywords.
    A Hormuz or naphtha headline is a score-5 trigger even before prices move.
    """
    print("[Reuters RSS] fetching...")
    import xml.etree.ElementTree as ET

    feed_url = "https://feeds.reuters.com/reuters/businessNews"
    try:
        r = requests.get(feed_url, headers=HEADERS, timeout=10)
        r.raise_for_status()
    except Exception as e:
        print(f"  [Reuters] ERROR: {e}")
        return

    hashes = load_hashes()
    h = content_hash(r.content)
    if hashes.get("reuters_rss") == h:
        print("  [Reuters] no new items")
        return
    hashes["reuters_rss"] = h
    save_hashes(hashes)

    try:
        root = ET.fromstring(r.content)
    except ET.ParseError as e:
        print(f"  [Reuters RSS] parse error: {e}")
        return

    for item in root.iter("item"):
        title = (item.findtext("title") or "").lower()
        desc  = (item.findtext("description") or "").lower()
        text  = title + " " + desc
        matched = [kw for kw in KEYWORDS if kw in text]
        if not matched:
            continue

        is_fm    = "force majeure" in text
        is_hormuz= "hormuz" in text or "strait" in text
        score = 5 if (is_fm or is_hormuz) else 3

        signal = {
            "source":   "reuters_rss",
            "polymer":  "ALL",
            "grade":    "macro",
            "price_inr": None,
            "headline": item.findtext("title"),
            "keywords": matched,
        }
        print(f"  [Reuters] score={score} | {item.findtext('title')[:80]}")
        maybe_trigger(score, signal)
